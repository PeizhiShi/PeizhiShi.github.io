# ============================================================

# We import:
# - numpy / pandas for data processing
# - matplotlib for plotting
# - scikit-learn for modelling and evaluation

import numpy as np
import pandas as pd
import matplotlib.pyplot as plt

from sklearn.model_selection import train_test_split
from sklearn.linear_model import LinearRegression, LogisticRegression
from sklearn.metrics import mean_absolute_error, r2_score, mean_squared_error
from sklearn.metrics import accuracy_score, confusion_matrix, classification_report, roc_auc_score

# Set a random seed for reproducibility
np.random.seed(42)


# ============================================================
# Load the real-world dataset
# ============================================================

# This step loads a CSV file containing house price data.
# The dataset includes mixed variable types (numeric + categorical).

csv_path = "house_prices.csv"
df = pd.read_csv(csv_path)

# Print first few rows + shape to confirm it loaded correctly
print(df.head())
print("\nLoaded data shape:", df.shape)

# Common mistakes:
# - Wrong file name or wrong folder -> FileNotFoundError
# - File loads but columns are different from what you expect -> check df.columns


# ============================================================
# Descriptive statistics
# ============================================================

print("\nColumn names:")

print("\nData types:")
print(df.dtypes)

print("\nSummary statistics (numeric columns):")
print(df.describe())

# Data visualisation and understanding

# ============================================================
# Correlation matrix (numeric features)
# ============================================================

# Correlation is a quick diagnostic:
# - Which numeric variables move together?
# - Does price correlate strongly with some predictors?
# - Are predictors strongly correlated with each other (multicollinearity)?

numeric_df = df.select_dtypes(include=[np.number])

plt.figure()
plt.imshow(numeric_df.corr(), cmap="coolwarm")
plt.colorbar()
plt.xticks(range(len(numeric_df.columns)), numeric_df.columns, rotation=90)
plt.yticks(range(len(numeric_df.columns)), numeric_df.columns)
plt.title("Correlation Matrix (Numeric Features)")
plt.tight_layout()
plt.show()

# Common mistakes:
# - Claiming causation from correlation
# - Using correlation to compare categorical variables (not meaningful)


# ============================================================
# Convert categorical variables to dummy variables
# ============================================================

# Linear regression and logistic regression require numeric inputs.
# We one-hot encode categorical variables into dummy columns (0/1).

# drop_first=True avoids the dummy variable trap:
# - Without drop_first, one category can be perfectly predicted from others (perfect multicollinearity)

categorical_cols = df.select_dtypes(include=["object"]).columns.tolist()
df_model = pd.get_dummies(df, columns=categorical_cols, drop_first=True)

print("\nColumns after dummy encoding:")
print(df_model.columns.tolist())


# ============================================================
# Prepare features and target (Regression)
# ============================================================

# Define:
# - y = target (price)
# - X = features (all other variables)

y = df_model["price"]
X = df_model.drop(columns=["price"])

# Split into training and test sets
# test_size=0.25 means 25% for testing
X_train, X_test, y_train, y_test = train_test_split(
    X, y, test_size=0.25, random_state=42
)

# Common mistakes:
# - Training and testing on same data -> overly optimistic performance
# - Changing random_state gives different results (not wrong, just non-reproducible)


# ============================================================
# Train baseline linear regression (RAW price)
# ============================================================

model_raw = LinearRegression()
model_raw.fit(X_train, y_train)

y_pred_raw = model_raw.predict(X_test)

mae_raw = mean_absolute_error(y_test, y_pred_raw)
r2_raw = r2_score(y_test, y_pred_raw)
rmse_raw = np.sqrt(mean_squared_error(y_test, y_pred_raw))

print("\nBaseline model (RAW price):")
print("MAE :", mae_raw)
print("RMSE:", rmse_raw)
print("R2  :", r2_raw)


# ------------------------------------------------------------
# Residual plot (RAW model)
# ------------------------------------------------------------

# Residual = actual - predicted
# Residual plots help diagnose:
# - heteroscedasticity (funnel shape)
# - non-linear patterns (curved residuals)
# - outliers (points far away)

residuals_raw = y_test - y_pred_raw

plt.figure()
plt.scatter(y_pred_raw, residuals_raw, alpha=0.6)
plt.axhline(0)
plt.xlabel("Predicted Price")
plt.ylabel("Residual (Actual - Predicted)")
plt.title("Linear Regression (RAW): Residual Plot")
plt.show()

# Common mistakes:
# - Mixing up residual definition (pred - actual vs actual - pred)
# - Ignoring residual patterns and only focusing on MAE/R2


# ------------------------------------------------------------
# Coefficients table (RAW model)
# ------------------------------------------------------------

effects_raw = pd.DataFrame({
    "feature": X.columns,
    "coefficient": model_raw.coef_
}).sort_values(by="coefficient", ascending=False)

print(effects_raw)

# Common mistakes:
# - Thinking coefficients are causal effects
# - Ignoring multicollinearity (coefficients can flip sign)


# ============================================================
# EXTRA: Classification with Logistic Regression (High vs Low price)
# ============================================================

# In many business contexts, we may not need the exact price.
# Sometimes the decision is categorical:
# - high price vs low price
# - high risk vs low risk
# - churn vs no churn

# ------------------------------------------------------------
# Step 1: Create a binary target (high/low) from price
# ------------------------------------------------------------

# We use median price as a threshold:
# - price >= median -> high_price (1)
# - price < median  -> low_price  (0)
#
# Why median?
# - The median is not sensitive to extreme values (outliers).

median_price = df_model["price"].median()
y_bin = (df_model["price"] >= median_price).astype(int)

X_cls = df_model.drop(columns=["price"])

X_train_c, X_test_c, y_train_c, y_test_c = train_test_split(
    X_cls, y_bin, test_size=0.25, random_state=42, stratify=y_bin
)

print("\nClassification threshold (median price):", median_price)
print("Class balance overall:")
print(y_bin.value_counts(normalize=True))

# Common mistakes:
# - Using a threshold that makes classes extremely imbalanced (e.g., top 5%)
# - Forgetting stratify -> train/test class ratio can shift


# ------------------------------------------------------------
# Step 2: Train Logistic Regression (NO Pipeline, NO StandardScaler)
# ------------------------------------------------------------

# Important note:
# - LogisticRegression can be trained directly on raw features.
# - Many tutorials use StandardScaler because it can help optimisation and
#   makes coefficients more comparable across different scales.
# - In this seminar we keep it simple and use raw features.
# - If you see a convergence warning, increase max_iter or consider scaling.

log_clf = LogisticRegression(max_iter=2000)
log_clf.fit(X_train_c, y_train_c)


# ------------------------------------------------------------
# Step 3: Evaluate classification performance
# ------------------------------------------------------------

# We evaluate with:
# - Accuracy: fraction of correct predictions

y_pred_c = log_clf.predict(X_test_c)
y_proba_c = log_clf.predict_proba(X_test_c)[:, 1]

acc = accuracy_score(y_test_c, y_pred_c)

print("\nLogistic Regression Results (RAW features):")
print("Accuracy:", acc)

# ------------------------------------------------------------
# Q2: Why do we convert categorical variables into dummy variables?
# ------------------------------------------------------------

# Task:
# Explain why we need dummy variables for categorical features.

# ANSWER (conceptual):
# - Linear regression requires numeric inputs.
# - Dummy variables represent categories as 0/1 indicators.
# - drop_first=True avoids perfect multicollinearity.

# Explanation:
# - If we keep categories as strings, the model cannot do arithmetic with them.
# - If we encode categories as 1/2/3, we accidentally create a fake ordering.

# Common mistakes:
# - Encoding categories as 1/2/3 directly (creates fake ordering)
# - Forgetting drop_first=True (dummy variable trap)

# ------------------------------------------------------------
# Q4: What does R2 tell us?
# ------------------------------------------------------------

# Task:
# Explain what R2 means in model evaluation.

# ANSWER (conceptual):
# - R2 is the proportion of variance in y explained by the model (on the test set).
# - R2 = 0 means no better than predicting the mean.
# - R2 can be negative if the model is worse than predicting the mean.

# Explanation:
# - R2 is useful, but it does not tell you the size of the error in currency units.
# - MAE/RMSE provide error size in the same unit as price.

# Common mistakes:
# - Using only R2 and ignoring MAE/RMSE
# - Comparing R2 across datasets with very different targets/features

# ------------------------------------------------------------
# Q7: What happens if we predict log(price) instead of price?
# Which model performs better, RAW or LOG?
# ------------------------------------------------------------

# Task:
# Compare the performance (MAE, RMSE and R2) of:
# - RAW model (predict price directly)
# - LOG model (predict log(price), then exp back)

# Hint: use np.log(...) to convert the price and np.exp(...) to convert back
y_train_log = np.log(y_train)


# Answer:
model_log = LinearRegression()
model_log.fit(X_train, y_train_log)

y_pred_log = model_log.predict(X_test)
y_pred_log_back = np.exp(y_pred_log)

mae_log = mean_absolute_error(y_test, y_pred_log_back)
r2_log = r2_score(y_test, y_pred_log_back)
rmse_log = np.sqrt(mean_squared_error(y_test, y_pred_log_back))

print("\nQ7 Answer:")
print("Advanced model (LOG price):")
print("MAE :", mae_log)
print("RMSE:", rmse_log)
print("R2  :", r2_log)

# ------------------------------------------------------------
# Residual plot (LOG model, back-transformed)
# ------------------------------------------------------------
residuals_log = y_test - y_pred_log_back

plt.figure()
plt.scatter(y_pred_log_back, residuals_log, alpha=0.6)
plt.axhline(0)
plt.xlabel("Predicted Price (from log model)")
plt.ylabel("Residual (Actual - Predicted)")
plt.title("Linear Regression (LOG): Residual Plot")
plt.show()


print("RAW model -> MAE:", mae_raw, ", RMSE:", rmse_raw, ", R2:", r2_raw)
print("LOG model -> MAE:", mae_log, ", RMSE:", rmse_log, ", R2:", r2_log)

# Explanation:
# - Lower MAE is better (smaller typical absolute error in currency units)
# - Lower RMSE is better (penalises large errors more strongly than MAE)
# - Higher R2 is better (explains more variance)

# House prices are often right-skewed.
# Taking log(price) compresses large values and often makes:
# - errors more stable
# - relationships closer to linear

# Common mistakes:
# - Evaluating log-model predictions without converting back using exp()
# - Thinking log transform always improves results (not guaranteed)
# - Comparing RMSE between different target scales (must be in same units)


# ------------------------------------------------------------
# Q8: If area increases by ONE unit, by approximately how many percent does price change?
# ------------------------------------------------------------

# Task:
# Use the LOG model coefficient to interpret the effect in percentage terms.

effects_log = pd.DataFrame({
    "feature": X.columns,
    "coefficient": model_log.coef_
}).sort_values(by="coefficient", ascending=False)

area_coef_log = effects_log.loc[effects_log["feature"] == "area", "coefficient"].values[0]


# ANSWER
approx_pct = area_coef_log * 100
exact_pct = (np.exp(area_coef_log) - 1) * 100

print("\nQ8 Answer:")
print("Area coefficient (LOG model):", area_coef_log)
print("Approx % change for +1 area:", approx_pct, "%")
print("Exact % change for +1 area:", exact_pct, "%")

# Explanation:
# If log(price) = ... + b*area + ...
# - A +1 increase in area multiplies price by exp(b)
# - Exact percent change = (exp(b) - 1)*100
# - Approx percent change ~ b*100 (only accurate if b is small)

# Common mistakes:
# - Treating b as an absolute currency change (wrong for log model)
# - Forgetting exp(b) interpretation
